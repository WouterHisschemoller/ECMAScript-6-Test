Default function parameters

Example shows default parameter use in later default parameters: 3 + (3 * 5) = 18.


function defPar(a = 3, b = a * 5) {
    return a + b;
}
document.getElementById('deffunpar').innerHTML = defPar();

Rest parameters

Example creates string of all parameters and shows the number of rest parameters.


function restPar(a, b, ...args) {
    var str = a + b;
    for(var i = 0; i < args.length; i++) {
        str += args[i];
    }
    return str + ' ' + args.length;
}
document.getElementById('restpar').innerHTML = restPar('W', 'o', 'u', 't', 'e', 'r');

for..of loops

Example shows difference between for..of and for..in loop with array ['one', 'two', 'three'].


var arr = ['one', 'two', 'three'];
var str = '';
for(var i of arr) {
    str += i + ' ';
}
str += '
';
for(var i in arr) {
    str += i + ' ';
}
document.getElementById('forof').innerHTML = str;

Tagged template string

Example shows manipulation of string template by a function


function tagFunction(strings, ...values) {
    return strings[0] + ' ' + strings[1] + ' ' + strings[2] + values[0] + values[1] + values[2];
}
